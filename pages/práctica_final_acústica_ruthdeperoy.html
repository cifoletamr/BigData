Utilizar los datos de instalaciones de red de monitorización del ruido ambiental en la ciudad de Barcelona para generar un modelo de visualización de estos y un modelo predictivo para estimar los datos de contaminación acústica durante el año 2024.

#Cambiamos la hora a float para que no nos de problemas en el análisis

def cambia_hora(row):
    cad = row['Hora']
    num = 0.0
    try : 
        cad = cad.replace(":",".").strip()
        num = float(cad)
    except :
        pass
      
    return num
    
df['HoraN'] = df.apply(cambia_hora, axis=1)

# Evaluar el modelo
